SUIT Construct

Available Since: TIE (0.0.0, 1.0.0, 1.1.0, 1.2.0) SUIT (1.3.0)

SUIT's Construct Function

Syntax object new SUIT ( array $config )
Parameters
config

An array of config we want to persist to $suit->config.

Return Values

The created SUIT object.

Examples
Basic Usage
Code <?php
require '/path/to/suit.class.php';
$config = array
(
    'files' => array
    (
        'code' => '/path/to/code',
        'templates' => '/path/to/templates'
    ),
    'filetypes' => array
    (
        'code' => 'inc.php',
        'templates' => 'tpl'
    ),
    'flag' => array
    (
        'insensitive' => true,
    ),
    'parse' => array
    (
        'attribute' => array
        (
            'equal' => '=',
            'separator' => ' ',
            'quote' => '"'
        ),
        'escape' => '\\',
        'section' => array
        (
            'open' => '[',
            'close' => ']',
            'end' => '/',
            'trim' => "\r.\n.\t ."
        ),
        'separator' => '=>'
    )
);
$suit = new SUIT($config);
$suit->config['parse']['nodes'] = array
(
    '[!' => array
    (
        'close' => '!]',
        'function' => array
        (
            array
            (
                'function' => 'templates',
                'class' => $suit->nodes
            )
        ),
        'var' => array
        (
            'escape' => $suit->config['parse']['escape'],
            'separator' => $suit->config['parse']['separator']
        )
    ),
    '[:' => array
    (
        'close' => ':]',
        'class' => $suit->nodes,
        'function' => array
        (
            array
            (
                'function' => 'variables',
                'class' => $suit->nodes
            )
        ),
        'var' => array
        (
            'escape' => $suit->config['parse']['escape'],
            'separator' => $suit->config['parse']['separator']
        )
    ),
    '[*' => array
    (
        'close' => '*]',
        'function' => array
        (
            array
            (
                'function' => 'comments',
                'class' => $suit->nodes
            )
        ),
        'skip' => true
    ),
    '[*' => array
    (
        'close' => '*]',
        'function' => array
        (
            array
            (
                'function' => 'comments',
                'class' => $suit->nodes
            )
        ),
        'skip' => true
    ),
    '[escape]' => array
    (
        'close' => '[/escape]',
        'function' => array
        (
            array
            (
                'function' => 'escape',
                'class' => $suit->nodes
            )
        ),
        'skip' => true,
        'skipescape' => true,
        'var' => $suit->config['parse']['section']['trim']
    ),
    $suit->config['parse']['section']['open'] . 'if' . $suit->config['parse']['section']['close'] => array
    (
        'close' => $suit->config['parse']['section']['open'] . $suit->config['parse']['section']['end'] . 'if' . $suit->config['parse']['section']['close'],
        'function' => array
        (
            array
            (
                'function' => 'condition',
                'class' => $suit->nodes
            )
        ),
        'skip' => true,
        'strip' => true,
        'var' => array
        (
            'condition' => false,
            'else' => false,
            'trim' => $suit->config['parse']['section']['trim']
        )
    ),
    $suit->config['parse']['section']['open'] . 'if ' => array
    (
        'close' => $suit->config['parse']['attribute']['quote'] . $suit->config['parse']['section']['close'],
        'function' => array
        (
            array
            (
                'function' => 'attribute',
                'class' => $suit->nodes
            ),
            array
            (
                'function' => 'conditionskip',
                'class' => $suit->nodes
            )
        ),
        'attribute' => $suit->config['parse']['section']['open'] . 'if' . $suit->config['parse']['section']['close'],
        'skip' => true,
        'skipignore' => true,
        'var' => array
        (
            'escape' => $suit->config['parse']['escape'],
            'equal' => $suit->config['parse']['attribute']['equal'],
            'separator' => $suit->config['parse']['attribute']['separator'],
            'quote' => $suit->config['parse']['attribute']['quote']
        )
    ),
    $suit->config['parse']['section']['open'] . 'loop' . $suit->config['parse']['section']['close'] => array
    (
        'close' => $suit->config['parse']['section']['open'] . $suit->config['parse']['section']['end'] . 'loop' . $suit->config['parse']['section']['close'],
        'function' => array
        (
            array
            (
                'function' => 'loop',
                'class' => $suit->nodes
            )
        ),
        'skip' => true,
        'var' => array
        (
            'vars' => serialize(array()),
            'delimiter' => '',
            'trim' => $suit->config['parse']['section']['trim'],
            'node' => array
            (
                'open' => '[|',
                'close' => '|]',
                'separator' => $suit->config['parse']['separator']
            )
        )
    ),
    $suit->config['parse']['section']['open'] . 'loop ' => array
    (
        'close' => $suit->config['parse']['attribute']['quote'] . $suit->config['parse']['section']['close'],
        'function' => array
        (
            array
            (
                'function' => 'attribute',
                'class' => $suit->nodes
            )
        ),
        'attribute' => $suit->config['parse']['section']['open'] . 'loop' . $suit->config['parse']['section']['close'],
        'skip' => true,
        'skipignore' => true,
        'var' => array
        (
            'blacklist' => true,
            'escape' => $suit->config['parse']['escape'],
            'equal' => $suit->config['parse']['attribute']['equal'],
            'list' => array('node'),
            'separator' => $suit->config['parse']['attribute']['separator'],
            'quote' => $suit->config['parse']['attribute']['quote']
        )
    ),
    '[return ' => array
    (
        'close' => '/]',
        'function' => array
        (
            array
            (
                'function' => 'returning',
                'class' => $suit->nodes
            )
        ),
        'skip' => true
    )
);
$suit->vars['condition'] = array();
$suit->vars['loop'] = array(); //Result: Creates the SUIT object
?>
Version Notes

In PySUIT, some default config entries are modified or removed. Here is the same example in PySUIT.

config 
import suit
config = {
    'files':
    {
        'code': '/home/brandon/pytemplates/code',
        'templates': '/home/brandon/pytemplates/templates'
    },
    'filetypes':
    {
        'code': 'py',
        'templates': 'tpl'
    },
    'flag':
    {
        'insensitive': True
    },
    'parse':
    {
        'attribute':
        {
            'equal': '=',
            'separator': ' ',
            'quote': '"'
        },
        'escape': '\\',
        'section':
        {
            'open': '[',
            'close': ']',
            'end': '/',
            'trim': '\r\n\t'
        },
        'separator': '=>'
    }
}
suitclass = suit.SUIT(config)
suitclass.config['parse']['nodes'] = {
    '[!':
    {
        'close': '!]',
        'function': [suit.nodes.templates],
        'var':
        {
            'escape': suitclass.config['parse']['escape'],
            'separator': suitclass.config['parse']['separator']
        }
    },
    '[:':
    {
        'close': ':]',
        'function': [suit.nodes.variables],
        'var':
        {
            'escape': suitclass.config['parse']['escape'],
            'separator': suitclass.config['parse']['separator']
        }
    },
    '[*':
    {
        'close': '*]',
        'function': [suit.nodes.comments],
        'skip': True
    },
    '[escape]':
    {
        'close': '[/escape]',
        'function': [suit.nodes.escape],
        'skip': True,
        'skipescape': True,
        'var': suitclass.config['parse']['section']['trim']
    },
    ''.join((
        suitclass.config['parse']['section']['open'],
        'if',
        suitclass.config['parse']['section']['close']
    )):
    {
        'close': ''.join((
            suitclass.config['parse']['section']['open'],
            suitclass.config['parse']['section']['end'],
            'if',
            suitclass.config['parse']['section']['close']
        )),
        'function': [suit.nodes.condition],
        'skip': True,
        'strip': True,
        'var':
        {
            'condition': False,
            'else': False,
            'trim': suitclass.config['parse']['section']['trim']
        }
    },
    ''.join((
        suitclass.config['parse']['section']['open'],
        'if '
    )):
    {
        'close': ''.join((
            suitclass.config['parse']['attribute']['quote'],
            suitclass.config['parse']['section']['close']
        )),
        'function':
        [
            suit.nodes.attribute,
            suit.nodes.conditionskip
        ],
        'attribute': ''.join((
            suitclass.config['parse']['section']['open'],
            'if',
            suitclass.config['parse']['section']['close']
        )),
        'skip': True,
        'skipignore': True,
        'var':
        {
            'escape': suitclass.config['parse']['escape'],
            'equal': suitclass.config['parse']['attribute']['equal'],
            'separator': suitclass.config['parse']['attribute']['separator'],
            'quote': suitclass.config['parse']['attribute']['quote']
        }
    },
    ''.join((
        suitclass.config['parse']['section']['open'],
        'loop',
        suitclass.config['parse']['section']['close']
    )):
    {
        'close': ''.join((
            suitclass.config['parse']['section']['open'],
            suitclass.config['parse']['section']['end'],
            'loop',
            suitclass.config['parse']['section']['close']
        )),
        'function': [suit.nodes.loop],
        'skip': True,
        'var':
        {
            'vars': pickle.dumps([]),
            'delimiter': '',
            'trim': suitclass.config['parse']['section']['trim'],
            'node':
            {
                'open': '[|',
                'close': '|]',
                'separator': suitclass.config['parse']['separator']
            }
        }
    },
    ''.join((
        suitclass.config['parse']['section']['open'],
        'loop '
    )):
    {
        'close': ''.join((
            suitclass.config['parse']['attribute']['quote'],
            suitclass.config['parse']['section']['close']
        )),
        'function': [suit.nodes.attribute],
        'attribute': ''.join((
            suitclass.config['parse']['section']['open'],
            'loop',
            suitclass.config['parse']['section']['close']
        )),
        'skip': True,
        'skipignore': True,
        'var':
        {
            'blacklist': True,
            'escape': suitclass.config['parse']['escape'],
            'equal': suitclass.config['parse']['attribute']['equal'],
            'list': ('node',),
            'separator': suitclass.config['parse']['attribute']['separator'],
            'quote': suitclass.config['parse']['attribute']['quote']
        }
    },
    '[return ':
    {
        'close': '/]',
        'function': [suit.nodes.returning],
        'skip': True
    }
}
suitclass.vars['condition'] = {}
suitclass.vars['loop'] = {} #Result: Creates the SUIT object

Also, suit.template is the variable in which the template is stored and returned.
