SUIT Construct

Available Since: TIE (0.0.0, 1.0.0, 1.1.0, 1.2.0) SUIT (1.3.0)

SUIT's Construct Function

Syntax object new SUIT ( array $config )
Parameters
config

An array of config we want to persist to $suit->config.

Return Values

The created SUIT object.

Examples
Basic Usage
Code <?php
function nodecomments()
{
    return '';
}

function nodetemplates($params)
{
    //Split up the file, paying attention to escape strings
    $split = $params['suit']->explodeunescape('=>', $params['case']);
    $code = array();
    foreach ($split as $key => $value)
    {
        //If this is the template file, get the file's template
        if ($key == 0)
        {
            $template = file_get_templates($params['suit']->config['files']['templates'] . '/' . $value . '.' . $params['suit']->config['filetypes']['templates']);
        }
        //Else, prepare to include the file
        else
        {
            $code[] = str_replace(array('../', '..\\'), '', $params['suit']->config['files']['code'] . '/' . $value . '.' . $params['suit']->config['filetypes']['code']);
        }
    }
    return $params['suit']->gettemplate($template, $code);
}

function nodevars($params)
{
    //Split up the file, paying attention to escape strings
    $split = $params['suit']->explodeunescape('=>', $params['case']);
    $var = $params['suit']->vars;
    foreach ($split as $value)
    {
        $var = $var[$value];
    }
    return $var;
}

require 'suit/suit.class.php';
$config = array

Version Notes

In PySUIT, some default config entries are modified or removed. Here is a copy of the standard config.

config 
"""Default config for SUIT along with node functions"""
def nodecomments():
    """Function used in default nodes"""
    return ''

def nodetemplates(params):
    """Function used in default nodes"""
    return params['suit'].gettemplate(params['case'])

def nodetemplates(params):
    #Split up the file, paying attention to escape strings
    split = params['suit'].explodeunescape('=>', params['case'])
    code = []
    for key, value in enumerate(split):
        #If this is the template file, get the file's template
        if key == 0:
            template = open(
                ''.join
                ((
                    params['suit'].config['files']['templates'],
                    '/',
                    value,
                    '.',
                    params['suit'].config['filetypes']['templates']
                ))
            ).read()
        #Else, prepare to include the file
        else:
            code.append(
                ''.join
                ((
                    params['suit'].config['files']['code'],
                    '/',
                    value,
                    '.',
                    params['suit'].config['filetypes']['code']
                )).replace('../', '').replace('..\\', '')
            )
	return params['suit'].gettemplate(template, code)

def nodevars(params):
    """Function used in default nodes"""
    #Split up the file, paying attention to escape strings
    split = params['suit'].explodeunescape('=>', params['case'])
    var = params['suit'].vars
    for value in split:
        var = var[value]
    return var
